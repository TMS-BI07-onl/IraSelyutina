SELECT Year, [January], [February], [March], [April], [May],
[June], [July], [August], [September],
[October], [November], [December]
FROM (SELECT DATENAME(year,[DueDate]) as year, DATENAME(month,[DueDate]) As month, [OrderQty] 
FROM Production.WorkOrder) AS SourceTable 
PIVOT
(SUM ([OrderQty]) FOR month IN (January, February, March, April, May,
June, July, August, September,
October, November, December)
) AS PivotTable
